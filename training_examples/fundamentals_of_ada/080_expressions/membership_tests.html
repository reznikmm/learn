Fri; Day : Calendar_Days := Calendar_Days'Val (X); begin if Day in Sun | Sat then -- identical expressions B := Day in Mon .. Fri; C := Day in Weekdays; Day := Wed; elsif Day = Mon or Day = Tues then D := D and (B or C); Day := Thur; end if; Put_Line (D'Image & " " & B'Image & " " & C'Image); Put_Line (Day'Image); end Membership_Tests;
